\newcommand{\drawTetrisGrid}[0]
{
 \foreach \y in {1,...,20}{
    \pgfmathsetmacro\yoffset{\y +1};
  	\foreach \x in {1,...,10}{
   	\pgfmathsetmacro\xoffset{\x +1};
  		\filldraw[fill=none, draw=black] (\x , \y ) 
  		  rectangle (\xoffset , \yoffset);
  	}
  }
}
