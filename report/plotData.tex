% Plot for means
\def \meansPlot 
{
\begin{axis}[%
	xmin=0,
	xmax=80,
    xlabel={Iteration/Generation},
    ylabel={Mean score},
    ymode=log,
    log basis y={10},
    width=\textwidth,
    height=12cm,
    legend pos=south east]
\addplot[color=black,mark=none, dotted, thick] 
    table [x expr=\thisrowno{0}, 
           y expr=\thisrowno{1}, 
           col sep=comma] {data/Cemeans/noNoise.txt};
\addplot[color=black,mark=none, thick] 
    table [x expr=\thisrowno{0}, 
           y expr=\thisrowno{1}, 
           col sep=comma] {data/Cemeans/constantNoise.txt};
\addplot[color=black,mark=none,dashed, thick] 
    table [x expr=\thisrowno{0}, 
           y expr=\thisrowno{1}, 
           col sep=comma] {data/Cemeans/linearNoise.txt};
\legend{No noise, constant noise, Linear decreasing noise}
\end{axis}
}
